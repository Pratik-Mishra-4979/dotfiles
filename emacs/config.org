;;; Performance: File-name-handler-alist

(defvar my--old-file-name-handler-alist (default-toplevel-value
					 'file-name-handler-alist))

(defun my--respect-file-handlers (fn args-left)
  "Respect file handlers.
FN is the function and ARGS-LEFT is the same argument as `command-line-1'.
Emacs processes command-line files very early in startup. These files may
include special paths like TRAMP paths, so restore `file-name-handler-alist' for
this stage of initialization."
  (let ((file-name-handler-alist (if args-left
				     my--old-file-name-handler-alist
				   file-name-handler-alist)))
    (funcall fn args-left)))

(defun my--restore-file-name-handler-alist ()
  "Restore `file-name-handler-alist'."
  (set-default-toplevel-value
   'file-name-handler-alist
   ;; Merge instead of overwrite to preserve any changes made since startup.
   (delete-dups (append file-name-handler-alist
			my--old-file-name-handler-alist))))


;; Determine the state of bundled libraries using calc-loaddefs.el. If
;; compressed, retain the gzip handler in `file-name-handler-alist`. If
;; compiled or neither, omit the gzip handler during startup for improved
;; startup and package load time.
(set-default-toplevel-value
 'file-name-handler-alist
 (if (locate-file-internal "calc-loaddefs.el" load-path)
     nil
   (list (rassq 'jka-compr-handler
		my--old-file-name-handler-alist))))

;; Ensure the new value persists through any current let-binding.
(put 'file-name-handler-alist 'initial-value
     my--old-file-name-handler-alist)

;; Emacs processes command-line files very early in startup. These files may
;; include special paths TRAMP. Restore `file-name-handler-alist'.
(advice-add 'command-line-1 :around #'my--respect-file-handlers)

(add-hook 'emacs-startup-hook #'my--restore-file-name-handler-alist
	  101)

;;; Performance: Disable mode-line during startup

(defvar-local my--hidden-mode-line nil
  "Store the buffer-local value of `mode-line-format' during startup.")

(when (not noninteractive)
  (put 'mode-line-format
       'initial-value (default-toplevel-value 'mode-line-format))
  (setq-default mode-line-format nil)
  (dolist (buf (buffer-list))
    (with-current-buffer buf
      (when (local-variable-p 'mode-line-format)
	(setq my--hidden-mode-line mode-line-format)
	(setq mode-line-format nil)))))

;;; Restore values
(defun my--startup-load-user-init-file (fn &rest args)
  "Ensure UI state is restored after startup, even on error."
  (unwind-protect
      ;; Run normal init loading
      (apply fn args)

    ;; Always restore message and redisplay
    (setq-default inhibit-message nil)
    (setq-default inhibit-redisplay nil)

    ;; Restore mode-line (global default)
    (unless (default-toplevel-value 'mode-line-format)
      (setq-default mode-line-format
		    (get 'mode-line-format 'initial-value)))

    ;; Restore mode-line in existing buffers
    (dolist (buf (buffer-list))
      (with-current-buffer buf
	(when (local-variable-p 'my--hidden-mode-line)
	  (setq mode-line-format my--hidden-mode-line)
	  (kill-local-variable 'my--hidden-mode-line))))))

(advice-add 'startup--load-user-init-file :around
	    #'my--startup-load-user-init-file)

#+begin_src emacs-lisp :tangle "~/.emacs.d/init.el"
(defvar emacs-solo-center-document-desired-width 100
  "The desired width of a document centered in the window.")

(defun emacs-solo/center-document--adjust-margins ()
  ;; Reset margins first before recalculating
  (set-window-parameter nil 'min-margins nil)
  (set-window-margins nil nil)

  ;; Adjust margins if the mode is on
  (when emacs-solo/center-document-mode
    (let ((margin-width (max 0
			     (truncate
			      (/ (- (window-width)
				    emacs-solo-center-document-desired-width)
				 2.0)))))
      (when (> margin-width 0)
	(set-window-parameter nil 'min-margins '(0 . 0))
	(set-window-margins nil margin-width margin-width)))))

(define-minor-mode emacs-solo/center-document-mode
  "Toggle centered text layout in the current buffer."
  :lighter " Centered"
  :group 'editing
  (if emacs-solo/center-document-mode
      (add-hook 'window-configuration-change-hook #'emacs-solo/center-document--adjust-margins 'append 'local)
    (remove-hook 'window-configuration-change-hook #'emacs-solo/center-document--adjust-margins 'local))
  (emacs-solo/center-document--adjust-margins))


(add-hook 'org-mode-hook #'emacs-solo/center-document-mode)

(defun my/gptel-prepare-buffer (preset &optional name)
  "Start or switch to a GPTel buffer and apply PRESET."
  (interactive
   (list (intern (read-string "Preset: "))
	 (let ((s (read-string "Buffer name (optional): ")))
	   (unless (string-empty-p s)
	     (intern s)))))
  (require 'gptel)
  (require 'gptel-agent)
  (let ((buffer (get-buffer-create (format "*gptel-%s*" (or name preset)))))
    (with-current-buffer buffer
      (unless (derived-mode-p 'org-mode)
	(org-mode))
      (unless gptel-mode
	(gptel-mode))
      (when (and (not (equal name 'ask))
		 (= (buffer-size) 0))
	(insert "* prompt:\n"))
      (setq-local gptel-backend my/gptel-backend)
      (gptel--apply-preset preset #'my-local-setter))
    (pop-to-buffer buffer)
    buffer))

;; gptel-ask for one-shot queries.
(defun my/gptel-ask ()
  "Ask gptel in a dedicated buffer."
  (interactive)
  (require 'gptel)
  (require 'gptel-agent)
  (let* ((prompt
	  (read-string
	   "Ask GPT: "
	   (and (use-region-p)
		(buffer-substring-no-properties
		 (region-beginning) (region-end)))))
	 (ask-buffer (my/gptel-prepare-buffer 'egpt 'ask)))
    (when (buffer-live-p ask-buffer)
      (with-current-buffer ask-buffer
	(setq-local gptel-org-branching-context t)
	(setq-local gptel-prompt-prefix-alist
	      (copy-tree gptel-prompt-prefix-alist))
	(setf (alist-get 'org-mode gptel-prompt-prefix-alist) "*Prompt:* \n")
	(org-overview)
	(goto-char (point-max))
	(insert "\n* Prompt: ")
	(goto-char (point-max))
	(insert prompt)))
    (gptel-request prompt
      :buffer ask-buffer
      :position (with-current-buffer ask-buffer (point-max))
      :stream gptel-stream
      :transforms gptel-prompt-transform-functions
      :fsm (gptel-make-fsm :handlers gptel-send--handlers)
      :system (default-value 'gptel--system-message))))

(defun my/gptel-display-full-window (buffer _alist)
  "Display BUFFER and make it occupy the whole frame."
  (let ((win (get-buffer-window buffer t)))
    (unless win
      (setq win (selected-window))
      (with-selected-window win

(defun my/is-image-file (file)
  (let ((ext (downcase (or (file-name-extension file) ""))))
    (member ext '("png" "jpg" "jpeg" "gif" "bmp" "tiff" "webp" "svg"))))

(defun my/org--ensure-image-dir ()
  (unless (file-directory-p my/org-image-folder)
    (make-directory my/org-image-folder t)))

(defun my/org--unique-file (filename)
  (let* ((base (file-name-base filename))
	 (ext  (file-name-extension filename))
	 (ts   (format-time-string "%Y%m%d_%H%M%S")))
    (format "%s_%s.%s" base ts ext)))

;;;; Image handling

(defun my/org-dnd-handle-image (uri)
  (let* ((file (dnd-get-local-file-name uri))
	 (filename (my/org--unique-file
		    (file-name-nondirectory file)))
	 (new-path (expand-file-name filename my/org-image-folder)))

    (my/org--ensure-image-dir)

    (call-process
     "magick" nil nil nil
     file
     "-fuzz" "20%"
     "-trim" "+repage"
     "-strip"
     new-path)

    (insert (org-link-make-string (concat "file:" new-path)))))

  (pcase (read-char-choice
	  "[c] clean+insert  [i] insert raw  [o] org default: "
	  '(?c ?i ?o))
    (?c (my/org-dnd-handle-image uri) t)
    (?i (my/org-dnd-insert-raw uri) t)
    (?o nil))) ;; fallback

;;;; Core interception

(defun my/dnd-handle-multiple-urls (orig &rest args)
  "Intercept DND in Org mode for images only."

  ;; IMPORTANT: reset state per event
  (setq my/dnd-already-handled nil)

  (let* ((window (nth 0 args))
	 (urls   (nth 1 args)))
    (with-current-buffer (window-buffer window)
      (if (and (derived-mode-p 'org-mode)
	       (= (length urls) 1))
	  (let* ((uri  (car urls))
		 (file (dnd-get-local-file-name uri)))
	    (if (and file (my/is-image-file file))
		(let ((handled (my/org-dnd-image-menu uri)))
		  (if handled
		      (setq my/dnd-already-handled t)
		    (apply orig args)))
	      (apply orig args)))
	(apply orig args)))))

;;;; Block Org fallback ONLY when handled

(defun my/org-dnd-block-fallback (orig &rest args)
  (if my/dnd-already-handled
      (progn
	(setq my/dnd-already-handled nil)
	nil) ;; block Org handler
    (apply orig args)))

(defvar pratik-org-src-temp-files nil
  "List of temp files created for org-src + eglot integration.")

(defun pratik-org-src--edit-advice (orig-fun &rest args)
  "Advice around `org-edit-special' to enable LSP in source block buffers."
  (let* ((info (org-babel-get-src-block-info)) ; Only valid in Org buffer
	 (lang (car info))
	 (params (nth 2 info))
	 (is-remote (file-remote-p default-directory))
	 (tangle (cdr (assoc :tangle params)))
	 (dir (or (cdr (assoc :dir params)) default-directory))
	 (ext (pcase lang
		("python" "py")
		("emacs-lisp" "el")
		("latex" "tex")
		(_ "tmp")))
	 (file (if (and tangle (not (string= tangle "no")))
		   (expand-file-name tangle dir)
		 (make-temp-file "org-src-eglot" nil (concat "." ext)))))
    ;; Tangle first before switching modes
    (org-babel-tangle '(4))

    ;; Now call the original edit function (switches to edit buffer)
    (apply orig-fun args)

    ;; Now we are in the special edit buffer (e.g., python-mode)
    ;; Set buffer-local variables for LSP/flycheck to work
    (setq-local buffer-file-name file)
    (setq-local default-directory dir)

    ;; Track temp files for later cleanup
    (unless (and tangle (not (string= tangle "no")))
      (push file pratik-org-src-temp-files)
      (add-hook 'kill-buffer-hook
		(lambda ()
		  (when (and buffer-file-name
			     (member buffer-file-name pratik-org-src-temp-files))
		    (delete-file buffer-file-name)
		    (setq pratik-org-src-temp-files
			  (remove buffer-file-name pratik-org-src-temp-files))))
		nil t))

    ;; Start LSP or syntax checker
    (pcase lang
      ("python" (eglot-ensure))
      ("emacs-lisp" (flymake-mode 1))
      (_ nil))))

;; Use advice instead of a hook
(advice-add 'org-edit-special :around #'pratik-org-src--edit-advice)
